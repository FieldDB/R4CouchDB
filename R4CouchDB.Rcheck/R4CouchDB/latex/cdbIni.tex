\HeaderA{cdbIni}{Ini function}{cdbIni}
\keyword{misc}{cdbIni}
\begin{Description}\relax
Function returns a list with some default settings
\end{Description}
\begin{Usage}
\begin{verbatim}
cdbIni()
\end{verbatim}
\end{Usage}
\begin{Details}\relax
The list:
\code{     cdb <- list(
                serverName = "",
                port = "5984",
                DBName="",
                newDBName="",
                id="",
                rev="",
                dataList=list(),
                design="",
                view="",
                queryParam="",
                date=toString( Sys.Date()),
                error="",
                res=""
                )
} is returned if the packages \code{library(RCurl)} and
\code{library(RJSONIO)} are successfully loaded.
\end{Details}
\begin{Value}
\begin{ldescription}
\item[\code{cdb}] see details for the default settings of this function

\end{ldescription}
\end{Value}
