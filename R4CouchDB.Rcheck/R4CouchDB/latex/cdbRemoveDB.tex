\HeaderA{cdbRemoveDB}{Function to remove a database.}{cdbRemoveDB}
\keyword{misc}{cdbRemoveDB}
\begin{Description}\relax
Removing a database means sending a http- "DELETE"- request to
\code{http://cdb\$serverName:cdb\$port/ ...}
\end{Description}
\begin{Usage}
\begin{verbatim}
cdbRemoveDB(cdb)
\end{verbatim}
\end{Usage}
\begin{Arguments}
\begin{ldescription}
\item[\code{cdb}] The \code{cdb} has to provide \code{cdb\$serverName},
\code{cdb\$port} and  \code{cdb\$DBName}

\end{ldescription}
\end{Arguments}
\begin{Details}\relax
In \code{cdb} maybe a entry \code{cdb\$delDBName} should be provided
for more explicit deleting respectively more secure removing.
\end{Details}
\begin{Value}
\begin{ldescription}
\item[\code{cdb}] The CouchDB answer is stored in \code{cdb\$res}. Any
problems on the R side are reportet in \code{cdb\$error}

\end{ldescription}
\end{Value}
