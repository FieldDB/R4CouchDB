\HeaderA{cdbDeleteDoc}{Deletes a document from a database}{cdbDeleteDoc}
\keyword{misc}{cdbDeleteDoc}
\begin{Description}\relax
With a given \code{cdb\$id} this function sends a http \code{"DELETE"}
request to the url \code{.../cdb\$id?rev=cdb\$rev}.
\end{Description}
\begin{Usage}
\begin{verbatim}
cdbDeleteDoc(cdb)
\end{verbatim}
\end{Usage}
\begin{Arguments}
\begin{ldescription}
\item[\code{cdb}] Beside \code{cdb\$serverName}, \code{cdb\$port} and
\code{cdb\$DBName} the \code{cdb\$id} must be given. R errors are
reported    in \code{cdb\$errors}

\end{ldescription}
\end{Arguments}
\begin{Value}
\begin{ldescription}
\item[\code{cdb }] The result of the delete request is stored in
\code{cdb\$res}(whatever this means).

\end{ldescription}
\end{Value}
